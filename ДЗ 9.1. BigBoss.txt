function findValueByKey(companyName, company) {
    if (company && company.name === companyName) {
        return company;
    }
    

    if (company && company.clients) {
        for (let client of company.clients) {
            let result = findValueByKey(companyName, client);
            if (result) {
                return result;
            }
        }
    }
    

    if (company && company.partners) {
        for (let partner of company.partners) {
            let result = findValueByKey(companyName, partner);
            if (result) {
                return result;
            }
        }
    }
    
    return null;
}
